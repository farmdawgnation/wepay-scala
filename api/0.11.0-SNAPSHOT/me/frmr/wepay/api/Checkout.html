case class Checkout(accountId: Long, shortDescription: String, type: String, amount: Double, checkoutId: Option[Long] = scala.None, longDescription: Option[String] = scala.None, payerEmailMessage: Option[String] = scala.None, payeeEmailMessage: Option[String] = scala.None, referenceId: Option[String] = scala.None, appFee: Option[Double] = scala.None, feePayer: Option[String] = scala.None, redirectUri: Option[String] = scala.None, callbackUri: Option[String] = scala.None, autoCapture: Option[Boolean] = scala.None, requireShipping: Option[Boolean] = scala.None, shippingFee: Option[Double] = scala.None, chargeTax: Option[Boolean] = scala.None, mode: Option[String] = scala.None, prefillInfo: Option[JObject] = scala.None, fundingSources: Option[String] = scala.None, authorization: Option[CheckoutAuthorization] = scala.None, state: Option[String] = scala.None) extends ImmutableWePayResource[Checkout, CheckoutResponse] with Product with Serializable

An instance of the Checkout class. Used to represent an actual exchange of funds between two parties.

accountId

The Account ID the checkout is associated with.

shortDescription

Short description of the purchase.

type

The type of transaction. One of: GOODS, SERVICE, DONATION, or PERSONAL.

amount

The amount of the trnasaction.

checkoutId

The ID assigned to the checkout by WePay. Your app should almost never set this value.

longDescription

A long description of the purchase.

payerEmailMessage

A message from you that will be included in the confirmation email to the payer.

payeeEmailMessage

A message from you that will be included in the confirmation email to the payee.

referenceId

The reference ID for the checkout. Should be unique per checkout per app.

appFee

The fee, in dollars and cents, your application will collect on this transaction. Limited to 20% of total amount.

feePayer

The person who pays transaction fee. One of "Payee" or "Payer". Defaults to "Payer".

redirectUri

The URI the user will be redirected to upon completing or canceling the checkout.

callbackUri

The URI that IPNs will be sent to.

autoCapture

Sets whether or not the payment should be captured instantly. Defaults to true.

requireShipping

If true, payer will be required to enter a shipping address. Defaults to false.

shippingFee

The fee for shipping.

chargeTax

Determines whether or not tax will be charged.

mode

The mode the checkout will be displayed in. One of "regular" or "iframe". Defaults to "regular".

prefillInfo

A JObject containing any information to prepopulate on WePay. Fields are: 'name', 'email', 'phone_number', 'address', 'city', 'state', 'zip'.

fundingSources

Setting to determine what funding sources are allowed. Values are "bank,cc", "bank", or "cc".

authorization

The authorization information for pre-authorized checkouts.

state

The state of the checkout.
